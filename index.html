  const fmtYen = (n) => `¥${Number(n||0).toLocaleString("ja-JP")}`;


  const formatDate = (iso) =>

    new Date(iso).toLocaleDateString("en-US", { year:"numeric", month:"short", day:"numeric" });


  const addDays = (d, days) => { const out = new Date(d); out.setDate(out.getDate()+days); return out; };

  const addMonths = (d, months) => { const out = new Date(d); out.setMonth(out.getMonth()+months); return out; };

  const addYears = (d, years) => { const out = new Date(d); out.setFullYear(out.getFullYear()+years); return out; };


  const computeNextBillingISO = (trialDays, frequency) => {

    const now = new Date();

    if (trialDays && trialDays > 0) return addDays(now, trialDays).toISOString();

    return (frequency === "yearly" ? addYears(now, 1) : addMonths(now, 1)).toISOString();

  };


  const digitsOnly = (s) => (s||"").split("").filter(ch => ch >= "0" && ch <= "9").join("");

  const stripSpaces = (s) => (s||"").split(" ").join("");


  const getSelectedProduct = () => products.find(p => p.id === selectedProductId) || null;

  const hostedUrl = (pid) => `https://hosted.komoju.com/subscribe/${pid}`;


  // IME-safe input binding:

  function bindImeSafeInput(inputEl, getValue, setValue, postRenderUpdate){

    inputEl.value = getValue();

    let composing = false;


    inputEl.addEventListener("compositionstart", () => { composing = true; });

    inputEl.addEventListener("compositionend", (e) => {

      composing = false;

      setValue(e.target.value);

      postRenderUpdate();

    });

    inputEl.addEventListener("input", (e) => {

      if (!composing) {

        setValue(e.target.value);

        postRenderUpdate();

      }

    });

  }

  function cardEl(title, subtitle, rightPillText) {

    const root = document.createElement("div");

    root.className = "card";


    const head = document.createElement("div");

    head.className = "cardhead";


    const left = document.createElement("div");

    left.innerHTML = `

      <h2 class="ctitle">${escapeHtml(title)}</h2>

      ${subtitle ? `<p class="csub">${escapeHtml(subtitle)}</p>` : ``}

    `;


    head.appendChild(left);


    if (rightPillText) {

      const right = document.createElement("div");

      right.className = "pill";

      right.textContent = rightPillText;

      head.appendChild(right);

    }


    const body = document.createElement("div");


    root.appendChild(head);

    root.appendChild(body);


    return { root, body };

  }


  function fieldEl(label, hint, renderControl) {

    const wrap = document.createElement("div");

    wrap.className = "field";

    const l = document.createElement("div");

    l.className = "label";

    l.textContent = label;

    wrap.appendChild(l);

    if (hint) {

      const h = document.createElement("div");

      h.className = "hint";

      h.textContent = hint;

      wrap.appendChild(h);

    }

    renderControl(wrap);

    return wrap;

  }


  function escapeHtml(s){

    return (s ?? "").toString()

      .replaceAll("&","&amp;")

      .replaceAll("<","&lt;")

      .replaceAll(">","&gt;")

      .replaceAll("\"","&quot;")

      .replaceAll("'","&#039;");

  }
